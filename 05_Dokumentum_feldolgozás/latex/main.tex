\section{Dokumentum-feldolgozás}
\label{sec:DokumentumFeldolgozas}

\begin{frame}{Dokumentum-feldolgozás}
	
	\begin{itemize}
		\item Napjaink korszerû kommunikációs lehetõségei mellett is az írásos dokumentáció maradt az elsõdleges információtárolási és továbbítási forma
		\item A legtöbb írott dokumentumot azonban ma már számítógéppel állítjuk elõ
		\item Ennek lehetõségeirõl fogunk most beszélni
		\item A dokumentumokkal kapcsolatos tevékenységek összessége a \textbf{dokumentum-feldolgozás}. Ez három fõ részre bontható:
		
		\begin{enumerate}
			\item dokumentum készítés
			\item dokumentum tárolás
			\item dokumentum visszakeresés
		\end{enumerate}
		
		\item A dokumentum-feldolgozáson belül elõször foglalkozzunk a \textbf{szöveg-feldolgozás}sal, ami a dokumentum szöveges részeinek létrehozásához, tárolásához és visszakereséséhez kapcsolódó tevékenységek összességét jelenti
	\end{itemize}
	
\end{frame}

\subsection{Szöveg-feldolgozás}
\label{sec:SzovegFeldolgozas}

\begin{frame}{Szöveg-feldolgozás}
	
	\begin{itemize}
		\item Az elõállított elektronikus dokumentumok összetettségétõl függõen három szintet különböztetünk meg:
		
		\begin{enumerate}
			\item text-editor
					\\ formázás nélküli, text fájlok elõállítása és kezelése, pl. programozói editorok, html szerkesztõk
					%forrásszövegek (*.txt, *.c, *.pas, *.html, *.tex, \dots)
			\item szövegszerkesztés
					\\ formázott, nyomtatásra is alkalmas dokumentumok elõállítása és kezelése, pl. Microsoft Office, OpenOffice
			\item kiadványszerkesztés, desktop publishing (DTP)
					\\ igényes, nyomdai minõségû dokumentumok elõállítása és kezelése, pl. \TeX, \LaTeX, MS Publisher, Adobe InDesign, PageMaker, QuarkXPress, Corel Ventura
		\end{enumerate}
		
		\item Nincs éles határ az egyes kategóriák között, de nagyon \\fontos, hogy tisztában legyünk a különbségekkel
	\end{itemize}
	
\end{frame}

\begin{frame}{Szöveg-feldolgozás}
	
	\begin{itemize}
		\item A text-editorokkal most nem foglalkozunk
		\item A hangsúlyt a szövegszerkesztés és a kiadványszerkesztés vizsgálatára helyezzük
		\item Hol a határ a kettõ között?
		\begin{itemize}
			\item a megbízhatóan kezelhetõ dokumentumok mérete szerint:
			
			\begin{itemize}
				\item szövegszerkesztõ: 10-20 oldal
				\item kiadványszerkesztõ: nincs korlát
			\end{itemize}
			
			\item a kezelhetõ dokumentumok összetettsége szerint:
			
			\begin{itemize}
				\item szövegszerkesztõ: 1 szövegfolyam beszúrt objektumokkal
				\item kiadványszerkesztõ: több keret (frame) kezelése, átfolyó dokumentumokkal (újság), tartalomjegyzék, tárgymutató, irodalomjegyzék, stb.
			\end{itemize}
			
			\item az alkalmazás eredeti célja szerint:
			
			\begin{itemize}
				\item szövegszerkesztõ: alapvetõen szövegszerkesztõ kiadványszerkesztõ funkciókkal
				\item kiadványszerkesztõ: kiadványszerkesztõnek tervezve
			\end{itemize}
			
		\end{itemize}
		
	\end{itemize}
	
\end{frame}

\begin{frame}[allowframebreaks,fragile]{Szövegszerkesztõk kevésbé ismert funkciói}
	
	\begin{enumerate}
		\item A szöveg tartalmi egységei:
		
		\begin{itemize}
			\item betû, karakter -- letter, character
			\item szó -- word: szóközökkel (white space) határolt egység
			\item bekezdés -- paragraph: Enter karakterekkel határolt egység 
			\item szakasz (fejezet) -- section: szakaszvége jelekkel határolt egység
			\item teljes dokumentum -- document
			\\ $\ $
			\item Figyelem! A sor és az oldal nem tartalmi egység, automatikusan változhat
		\end{itemize}
		
		\item A szöveg formázása a tartalmi egységekhez igazodik:
		
		\begin{itemize}
			\item karakterjellemzõk: betûtípus, betûméret, betûstílus, betûszín, effektusok
			\item külön szó jellemzõk nincsenek, esetleg az elválasztást sorolhatjuk ide
			\item bekezdésjellemzõk: igazítás, térköz, sorköz, számozás, behúzás, tabulátorok, tördelés szabályozása
			\item szakaszjellemzõk: ide tartoznak a lap beállításai, azaz egy szakasz mindig azonos lapokból áll, lapméret, margók, tájolás, hasábok, fejléc, lábléc, oldalszámozás
		\end{itemize}
		
		\item Stílusok
		
		\begin{itemize}
			\item különbözõ szoftverek eltérõ módon teszik elérhetõvé a stílusok használatát, pl. MS-Office és OpenOffice: stílusok, HTML: CSS
			\item nem egyedileg formázzuk a betûket, bekezdéseket, helyette betû-, bekezdés-típusokat alakíthatunk ki
			\item egyszer meghatározzuk az egyes típusok jellemzõit
			\item az egyes szövegrészekrõl azt mondjuk meg, hogy milyen típusba tartozzon
			\item a konkrét formázást a szoftvert végzi el dinamikusan
			\item ha módosítjuk egy típus jellemzõit, akkor automatikusan megváltozik az összes ilyen típusú szövegrészlet formázása
			\item segíti, hogy mindig egységes maradjon a szöveg(ek) formázása
		\end{itemize}
		
		\item Sablonok
		
		\begin{itemize}
			\item mintafájl hasonló dokumentumok létrehozására
			\item szövegelemek és stílusok egyaránt lehetnek benne
			\item például egyszer kell elkészíteni egy szakdolgozat sablont, és mindenki használhatja ugyanazt
		\end{itemize}
				
		\item Nyelv
		
		\begin{itemize}
			\item napjainkban már egyértelmûen elvárás, hogy minden értelemben hibátlan dokumentumokat hozzunk létre bármilyen nyelven
			\item ez nem csak a begépelt szövegek nyelvét érinti
			\item helyesírási könyvtárak
			\item elválasztás
			\item írásjelek, tizedespont-vesszõ, idézõjelek
			\item standard szövegek: fejezet, ábra, kép, oldalszám, \dots
			\item dátum, idõ, pénznem formátumok
		\end{itemize}
		
		\item Automatikus javítás
			
			\begin{itemize}
			  \item segíti és gyorsítja a munkát, de bizonyos funkciókat érdemes kikapcsolni
				\item kikapcsolandó funkciók: KÉt KEzdõ NAgybetû, Cellák elsõ betûje nagybetû
			\end{itemize}
			
		\item Gyorsszöveg, Kész szöveg, Autotext, makrók
		
		\begin{itemize}
			\item gyakran begépelendõ szövegrészek gyorsítása
		\end{itemize}
		
		\item Helyesírás-ellenõrzés
		
		\begin{itemize}
			\item nagy segítség, de ne higgyünk neki vakon
			\item legyen kéznél egy megbízható helyesírási szótár, pl. http://magyarhelyesiras.hu/
		\end{itemize}
		
		\item Korrektúra
		
		\begin{itemize}
			\item nagyon hasznos lehetõség, ha egy nagyobb dokumentumon hosszabb ideig dolgozunk, több változat készül belõle, vagy többen dolgozunk rajta
			\item amikor egy már lezárt változaton módosítunk, vagy a szerzõn kívül valaki más szerkeszti a dokumentumot, akkor kapcsoljuk be a \emph{Korrektúra / Változatok követése} funkciót!
			\item így a módosításokat javaslatként rögzíti a rendszer, késõbb visszanézhetjük ezeket, és eldönthetjük, hogy elfogadjuk, vagy visszavonjuk ezeket
			\item legkésõbb a dokumentum lezárásakor ne felejtsünk el véglegesíteni minden javaslatot
			\item ha valaki átnézi a munkánkat, kérjük meg, hogy õ is így jelezze a javaslatait
		\end{itemize}
		
		\item Tabulátorok:
		
		\begin{itemize}
			\item a szóközzel történõ szövegigazítás helyett
			\item különbözõ igazítású tabulátorok: balra, jobbra, középre, tizedesvesszõ, \dots
			\item a bekezdéshez tartozó jellemzõ
			\item professzionális kinézetû eredményt ad kis munkával
		\end{itemize}
		
		\item Táblázatok:
		
		\begin{itemize}
			\item a tabulátorok mellett egy másik lehetõség táblázatok létrehozására
			\item nem feltétlenül kell behúzni a cellák közötti vonalakat
			\item használhatunk különbözõ vonaltípusokat is
			\item kombinálhatjuk a tabulátorokkal, de egy cellában Ctrl-Tab-bal tudunk tabulátorjelet beszúrni
		\end{itemize}
		
				\item Fejléc -- Lábléc -- (Oldalszámozás)
				
				\begin{itemize}
					\item szakaszjellemzõ, tehát egy szakaszon belül egyformának kell lennie
					\item ha különbözõt szeretnénk, akkor új szakaszt kell kezdenünk
					\item egy szakaszon belül is háromféle lehet: elsõ oldal, páros, páratlan oldal
					\item beszúrhatunk speciális mezõket is: dátum, oldalszám, oldalak száma, fejezetcímek, \dots
				\end{itemize}
				
				\item Címsorszintek
				
				\begin{itemize}
					\item különbözõ szintû alpontjaink lehetnek: 1-9
					\item MS-Office, OpenOffice: stílusok segítségével \\állíthatjuk be: Címsor1, Címsor2, \dots
					\item \LaTeX: külön parancsok segítségével állíthatjuk be: \verb|\part, \section, \subsection|, \dots
					\item ezen alapszik a tartalomjegyzék és a vázlatnézet
				\end{itemize}
				
				\item Vázlatnézet
				
				\begin{itemize}
					\item segíti a hosszabb dokumentumok áttekintését
					\item kiemeli a címsorokat
					\item szabályozhatjuk, hogy milyen mélységig akarjuk látni a címeket és az alattuk lévõ normál szövegeket
				\end{itemize}
				
				\item Körlevél készítés, címkék, borítékcímzés
				
				\begin{itemize}
					\item hasonló tartalmú, de eltérõ adatokat tartalmazó dokumentum sorozatok készítése
					\item különválasztjuk az adatokat és a sémát
					\item készítünk egy adatbázist, egy-egy rekord tartozik egy példányhoz
					\item készítünk egy sémát, amiben az állandó szövegrészek mellett megjelöljük az adatbázisból kiolvasandó adatelemeket
					\item összefuttatjuk a sémát és az adatforrást, így megkapjuk a dokumentumköteget
					\item a végeredmény lehet: különálló lapok, borítékok, különbözõ típusú etikettek (egy oldalon több címke)
				\end{itemize}
				
				\item Matematikai formulák kezelése
				
				\begin{itemize}
					\item tudományos dolgozatok esetében (a szakdolgozat is ilyen) kötelezõ elvárás, hogy megfelelõ minõségben tartalmazzon matematikai kifejezéseket
					\item a használandó eszköz kiválasztásakor vegyük ezt figyelembe
					\item több lehetõség közül választhatunk:
					
					\begin{enumerate}
						\item integrált képletszerkesztõ (MS-Office, OpenOffice)
						\item integrált matematikai leíró nyelv (\LaTeX)
						\item külsõ eszközökkel elõállított képletek képként beszúrva (Maple, Mathematica, WolframAlpha, \dots)
					\end{enumerate}
					
					\item bármelyik megoldást választjuk, szánjunk idõt az eszköz használatának megtanulására
					\item alaposan teszteljük le, hogy milyen módon fogjuk illeszteni a képleteket a szöveg többi részéhez, felbontás, oldalszámozás, képletsorszámozás, nyomtatási minõség, \dots
					\item tartsuk be a matematikai szövegek szedésére vonatkozó szabályokat, és megállapodásokat:
					
					\begin{itemize}
						\item speciális szimbólumok: $\sum$, $\prod$, $\int$, $\pm\infty$, \dots
						\item speciális mûveleti jelek: $\times, \forall, \exists, \notin, \cap, \subseteq$, \dots
						\item speciális ábécék: $\alpha, \beta, \Delta, \Omega, \Re$, \dots
						\item különbözõ fajta és méretû zárójelek: \\$\left(, \left[, \left\{, \left|, \right|, \right), \right], \right\}$, \dots
					\end{itemize}
					
					\item Például
					$$
					f(x) = \sum_{i=0}^{\infty}{\lambda_i x^i \left(\sqrt{\frac{x}{i!}}+\pi^2\right)}
					$$
					
				\end{itemize}
				
	\end{enumerate}
	
\end{frame}

\begin{frame}[allowframebreaks]{Tipográfiai tanácsok, ajánlások}
	
	\begin{enumerate}
		\item A dokumentumkészítés javasolt sorrendje: gépelés $\rightarrow$ formázás, a gépelés közben ne foglalkozzuk a formázással
		\item ENTER csak a bekezdés végén! Esetleg új sor karakter (Shift-Enter)
		\item Szóközbõl egyszerre csak egyet használunk, azaz szóközzel nem csinálunk helyet! Helyette tabulátorokat használjunk
		\item Nyitózárójel elõtt van szóköz (1 darab), utána nincsen
		\item Végzárójel elõtt nincs szóköz, utána van (1 darab)
		\item Többféle ,,kötõjel'' létezik: -, --, ---
		\item Többféle szóköz létezik: törhetõ, nem törhetõ, \\nyújtható, nem nyújtható
		\item Egyéb írásjelek elõtt nincsen szóköz, utána 1 darab \\(.:,;!?)
		\item A magyar idézõjel nem azonos az aposztróffal: \rmfamily ,,magyar idézet'', "aposztróf" \sffamily
		\item Ügyeljünk a hosszú szavak elválasztására, különösen akkor, ha sorkizárást használunk, használjunk automatikus, vagy opcionális elválasztást (Ctrl+kötõjel)
		\item Ügyeljünk a laptörésekre! Bizonyos pontokon nem lenne jó, ha odakerülne a laptörés. Ne üres sorokkal oldjuk meg a problémát! Miért?
		\\ A bekezdésjellemzõk között megadhatunk erre vonatkozó szabályokat: fattyú- és árvasorok, együtt a következõvel, \\egy oldalra, új oldalra
		\framebreak
		\\Például:

\begin{frame}{Mirõl is van szó matematikailag?}
	
	\begin{enumerate}
		\item Elsõfokú, lineáris Bezier-görbe:
		$$B(t) = P_0 + t(P_1 - P_0) = (1-t)P_0 + tP_1, t\in[0,1]	$$
		\item Másodfokú, quadratikus Bezier-görbe:
		$$B(t) = (1-t)^2P_0 + 2(1-t)tP_1 + t^2P_2, t\in[0,1]   $$
		\item Harmadfokú, köbös Bezier-görbe:
		$$B(t) = (1-t)^3P_0 + 3(1-t)^2tP_1 + 3(1-t)t^2P_2 + t^3P_3, t\in[0,1]   $$
		\item n-edfokú Bezier-görbe:
		$$B(t) = \sum_{i=0}^n{\binom{n}{i}(1-t)^{n-i}t^iP_i}    $$
	\end{enumerate}
	
\end{frame}

\begin{frame}{Additív és szubtraktív színkeverés}
	
			\begin{enumerate}
				\item Additív (összeadó) színkeverés
				
				\begin{itemize}
					\item alapszínek: Red-piros, Green-zöld, Blue-kék (RGB)
					\item a színek összeadódnak
					\item egy újabb szín hozzáadása növeli a színerõt
					\item a három alapszín egyenlõ arányú keveréke fehéret eredményez
					\item a színek hiánya fekete színt ad
					\item mintha újabb és újabb színes lámpákat kapcsolnánk be
				\end{itemize}
				
				\item Szubtraktív (kivonó) színkeverés
				
				\begin{itemize}
					\item alapszínek: Cyan-ciánkék, Magenta-bíborvörös, Yellow-sárga (CMY)
					\item az elnyelt színek szerint keverednek
					\item újabb szín bekeverése csökkenti a színerõt
					\item az alapszínek egyenlõ arányú keveréke feketét eredményez
					\item a színek hiánya fehéret ad
					\item mintha újabb és újabb átlátszó színes fóliákat \\helyeznénk egymásra
				\end{itemize}
				
			\end{enumerate}
	
\end{frame}

\begin{frame}[allowframebreaks]{Színek kódolása}
	Különbözõ módszerekkel kódolhatjuk a színeket:
	
	\begin{enumerate}
		\item Black and White (fekete-fehér):
		
		\begin{itemize}
			\item egy képpontnak két állapota van: fekete vagy fehér
			\item minden pixel 1 biten kódolható
		\end{itemize}
		
		\item 16 Color (16 szín, vagy szürkeárnyalat)
		
		\begin{itemize}
			\item minden pixel 4 biten kódolható
		\end{itemize}
		
		\item 256 Color (256 szín, vagy szürkeárnyalat)
		
		\begin{itemize}
			\item minden pixel 8 biten (1 bájt) kódolható
		\end{itemize}
		
		Az eddigi módszerek nem színkeverést kódolnak, hanem egy elõre definiált palettáról választják ki az egyik színt, \\a paletta kicserélésével automatikusan megváltoznak \\a kép színei
		\framebreak
		
		\item RYB színrendszer
		
		\begin{itemize}
			\item alapszínek: piros, sárga, kék, a tiszta színek
			\item másodlagos színek: narancs, zöld, lila
			\item szubtraktív rendszer
			\item fehér: színek hiánya
			\item a három alapszín összege: barna
			\item ezzel a rendszerrel tiszta fekete nem keverhetõ ki
		\end{itemize}
				
		\begin{figure}
			\centering
				\includegraphics[height=0.40\textheight]{images/ryb.png}
			\label{fig:ryb}
		\end{figure}
		\framebreak
		
		\item RGB színrendszer (24 bit Color), True Color
		
		\begin{itemize}
			\item alapszínek: piros, zöld, kék
			\item minden színkomponenst 256 fokozatban adhatunk meg
			\item 24 bit, 16 millió színárnyalat, 1 pixel: 3 bájt
			\item additív rendszer $\Rightarrow$ minden: fehér, semmi: fekete
			\item ezt használják a monitorok és a projektorok
		\end{itemize}
		
		\begin{figure}
			\centering
				\includegraphics[height=0.40\textheight]{images/rgb.png}
			\label{fig:rgb}
		\end{figure}
		\framebreak
		
		\item CMYK színrendszer (32 bit Color)
		
		\begin{itemize}
			\item alapszínek: türkiz (cyan), bíbor (magenta), sárga (yellow), fekete (blacK)
			\item minden színkomponenst 256 fokozatban adhatunk meg
			\item 32 bit, 4,3 milliárd (billió) színárnyalat, 1 pixel: 4 bájt
			\item szubtraktív rendszer $\Rightarrow$ minden: fekete, semmi: fehér
			\item ezt használják a nyomtatók
			\item egyes képszerkesztõ programok csak 0-100 közötti értéket engednek meg komponensenként, ez 100 millió színárnyalatot eredményez
		\end{itemize}
		
		\begin{figure}
			\centering
				\includegraphics[height=0.25\textheight]{images/cmyk.png}
			\label{fig:cmyk}
		\end{figure}
		\framebreak
		
		\item CMY színrendszer
		
		\begin{itemize}
			\item ugyanaz, mint a CMYK, csak fekete nélkül
			\item a fekete nem keverhetõ ki, csak sötétbarba
			\item szubtraktív rendszer $\Rightarrow$ minden: fekete, semmi: fehér
			\item ezt használják egyes tintasugaras nyomtatók, a fekete külön patron
			\item az RGB--CMY átalakítás egy vektormûvelettel egyszerûen megfogalmazható:
			$$
			\left[
			\begin{array}{c}
			C\\
			M\\
			Y\\
			\end{array}
			\right]
			=
			\left[
			\begin{array}{c}
			1\\
			1\\
			1\\
			\end{array}
			\right]
			-
			\left[
			\begin{array}{c}
			R\\
			G\\
			B\\
			\end{array}
			\right]
			$$

		\end{itemize}
		
		\framebreak

		\item HSB színrendszer
		
		\begin{itemize}
			\item három komponens:  Hue -- színárnyalat, Saturation -- telítettség, Brightness -- fényesség
			\item a színárnyalat egy színt választ ki a színkeréken 0 és 359 között
			\item a telítettség és a fényesség megadása százalékban történik
			\item a telítettség jelentése: mennyire  keskeny  sávot  határoz  meg  a  színkerékbõl. Nagyobb   érték   esetén   a   megadott   szín   távolabbi 
szomszédai  is  részt  vesznek  a  szín  kikeverésében,  a szín  pasztell, majd  szürkés  árnyalatú  lesz.  A  szín  a telítettség minimális értéke esetén ,,tiszta'' lesz.
		\end{itemize}
				
		\begin{figure}
			\centering
				\includegraphics[height=0.25\textheight]{images/szinkerek_alap.jpg}
			\label{fig:szinkerek_alap}
		\end{figure}
		\framebreak
		
		\item Színkerék modell tiszta színekkel
		
		\begin{itemize}
			\item a kerék peremén szerepelnek a színek
			\item egy egyenlõ oldalú háromszög csúcsaiban a tiszta színek: (kék, sárga, piros)
			\item velük átellenben a másodlagos színek (narancs, zöld, lila)
			\item két tetszõleges szín között a két színnel való keverés árnyalatai szerepelnek, félúton a két adott szín egyenlõ arányú keverésével kapott szín
			\item a színek sorrendje adja a szivárvány színsorrendjét
		\end{itemize}
				
		\begin{figure}
			\centering
				\includegraphics[height=0.25\textheight]{images/szinkerek_tiszta.jpg}
			\label{fig:szinkerek_tiszta}
		\end{figure}
		\framebreak
		
		\item Színkerék modell alap színekkel (RGB)
		
		\begin{itemize}
			\item a színkerék egy sugarán haladva az adott szín telítettsége változik
			\item a színkerék kerületén a tiszta színek találhatóak, a belsejében a pasztell színek
			\item ahogy ezen a sugáron a színkerék kerületétõl a középpont felé haladunk, az eredeti (kerületen lévõ) szín egyre nagyobb környezetébõl keverednek össze a színek és ez a keveredés adja a sugár adott helyén lévõ színárnyalatot. Maga a szín nem változik a sugár vonalában (mivel a kerületen lévõ szín jobb és bal oldaláról azonos mértékben adódnak a bekeveredõ színek és ezek a színek párosíthatóak egymással, a párok összege viszont az eredeti színt adja)
			\item a sugár belseje felé haladva a szín veszít a \\telítettségébõl
			\item minden ilyen sugár esetében a színkerék középpontjához érve a fehér színt kapjuk, mivel ekkor már az eredeti szín bõvülõ környezete eléri a színkerék teljes kerületét, az additív színek összessége viszont fehér színt eredményez.
		\end{itemize}
				
		\begin{figure}
			\centering
				\includegraphics[height=0.40\textheight]{images/szinkerek_alap.jpg}
			\label{fig:szinkerek_alap2}
		\end{figure}
		
	\end{enumerate}
	
\end{frame}

\subsection{Konverziók}
\label{sec:Konverziok}

\begin{frame}{Konverziók a raszter- és a vektorgrafika között}
	
	\begin{enumerate}
		\item Vektorgrafika $\rightarrow$ Rasztergrafika
		
		\begin{itemize}
			\item ez a gyakoribb és egyszerûbb eset
			\item majdnem mindig megtörténik, amikor egy program megjelenít egy vektorgrafikus ábrát
			\item a megadott sorrendben és szabályok szerint sorban ki kell rajzolni az objektumokat
		\end{itemize}
		
		\item Rasztergrafika $\rightarrow$ Vektorgrafika
		
		\begin{itemize}
			\item ez sokkal ritkább és nehezebb
			\item mesterséges intelligenciát igényel, hogy a pixelgrafikus ábrában felismerjük az összetartozó pixeleket
			\item speciális eset az optikai karakterfelismerés (OCR), a felismerendõ objektumok egy elõre meghatározott \\készletbõl kerülnek ki, az ábécé betûi lehetnek
		\end{itemize}
		
	\end{enumerate}
	
\end{frame}

\begin{frame}{Konverziók a fájlformátumok között}
	
	\begin{itemize}
		\item Miért fontos a konverzió?
		\item A különbözõ fájlformátumok eltérõ lehetõségekkel, szabályokkal, és korlátokkal rendelkeznek
		\item Ezek között vannak áthidalható, áthidalhatatlan és részben áthidalható ellentétek
		\item A konverzió speciális esete a tömörítés és a kódolás
		\item Különbözõ konverziók lehetségesek:
		
		\begin{itemize}
			\item megfordítható $\Leftrightarrow$ nem megfordítható
			\item veszteséges $\Leftrightarrow$ veszteségmentes
			\item minõségtartó, minõségcsökkentõ, minõségjavító
		\end{itemize}
		
		\item Fájlformátumok adatbázisa: http://www.wotsit.org/
	\end{itemize}
	
\end{frame}

\section{Verziókezelés}
\label{sec:Verziokezeles}

\begin{frame}{Verziókezelés}
	
	Mirõl van szó?
	
	\begin{itemize}
		\item különbözõ verziók, változatok születnek egyes dokumentumokból
		\item egy ember hosszabb ideig dolgozik rajta
		\item több ember dolgozik rajta egy idõben, vagy egymás után
		\item sokszor hasznos lenne, ha
		
		\begin{itemize}
			\item elõ tudnánk venni egy korábbi verziót
			\item elõ tudnánk venni valamelyik munkatárs által készített verziót
			\item össze tudnánk hasonlítani különbözõ verziókat
		\end{itemize}
		
		\item ezeket az igényeket tudják kielégíteni a verziókezelõ programok (Revision vagy Version Control System, RCS, VCS)
	\end{itemize}
	
\end{frame}

\begin{frame}{A munkafolyamat}
	
	\begin{itemize}
		\item elkészítem a dokumentum elsõ változatát
		\item átadom a verziókezelõ programnak (Check-in)
		\item a rendszer elteszi az adatbázisába (1.0 verzió)
		\item Ha legközelebb módosítani akarom a dokumentumot:
		\begin{itemize}
			\item a verziókezelõ programtól elkérem a dokumentum aktuális, vagy bármelyik korábbi verzióját (Check-out)
			\item módosítom a kapott dokumentumot
			\item az adatbázisban foglalttá, módosítottá (Modified) válik
			\item visszaadom a dokumentumkezelõ programnak (Check-in)
			\item a rendszer elteszi az adatbázisba, növeli a verziószámot, de megõrzi az elõzõ verziót is
			\item minden változathoz megjegyzések és további információk fûzhetõk
			\item mindez mûködik több felhasználó esetén is
		\end{itemize}
		\item Például: CS-RCS, Subversion, Git, Visual SourceSafe, \\GNU Arch, \dots
	\end{itemize}
	
\end{frame}
