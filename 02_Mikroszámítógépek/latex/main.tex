\begin{frame}{Regiszterek}
	
	\begin{itemize}
		\item Gyors mûködésû átmeneti tárolók:
	
		\begin{itemize}
			\item $8\dots512$ db 
			\item méretük különbözõ lehet: 4 bit $\dots$ 64 bit, ezt nevezzük a processzor szóhosszúságának
			\item az operatív memóriánál akár 100-szor gyorsabb lehet
			\item közvetlenül ezekkel dolgozik a processzor, a memóriából betölti az adatokat a regiszterekbe, feldolgozza, visszatölti a memóriába
		\end{itemize}

		\item Osztályozásuk:
		
		\begin{description}
			\item[Rendszer regiszterek] a felhasználó (programok) közvetlenül nem fér hozzá, pl. flag regiszter (állapotjelzõ), címbusz regiszter, adatbusz regiszter
			
			\item[Általános célú regiszterek] a programok használják, pl. akkumulátor, utasítás regiszter, utasításszámláló, címregiszter, adatregiszter
		\end{description}
		
	\end{itemize}
	
\end{frame}

\begin{frame}[allowframebreaks]{Flag regiszter}
	
	\begin{itemize}
		\item minden bitje egy-egy rendszerjellemzõt tárol, pl. az utoljára végrehajtott utasításban volt-e túlcsordulás, 0 volt-e az eredménye, mi volt az elõjele, stb. 
		\item bizonyos utasítások a flag regiszter bitjeinek állapotától függõen csinálnak valamit
		\item Például: \textbf{JZ >>cím<<}
		
		\begin{itemize}
			\item ,,Jump if zero'' (ugrás, ha nulla)
			\item a gépi kódú utasítások végrehajtásakor a processzor mindig beállítja a flag regiszter egyes bitjeit
			\item pl. ha az utoljára végrehajtott gépi kódú utasítás eredménye nulla volt, akkor egyre állítja zéró bitet
			\item ha a következõ utasítás a \textbf{JZ >>cím<<}, akkor ha a zéró bit értéke 1, akkor elugrik a >>cím<< memóriacímre és onnan folytatja a végrehajtást, egyébként folytatja a következõ utasításnál
			\item ez felel meg a magasszintû programozási nyelvek \textbf{IF} utasításának
			\item több hasonló utasítás létezik: 
		\end{itemize}
		
	\end{itemize}
		
	\begin{table}[htbp]
		\centering
			\begin{tabular}{|l|l|}
			\hline
			JZ (JE) & ugrás, ha egyenlõ (nulla) (equal-zero) \\
			JNZ (JNE) & ugrás, ha nem egyenlõ (nem nulla) (nonequal-nonzero) \\
			JG & ugrás, ha nagyobb (greater) \\
			JNG & ugrás, ha nem nagyobb (non-greater) \\	
			JA & ugrás, ha elõjel nélkül nagyobb \\
			JC & ugrás, ha elõjel nélkül kisebb \\
			\hline
			\end{tabular}
	\end{table}
	
\end{frame}

\begin{frame}{Aritmetikai-logikai egység (ALU)}
	Funkciói:
	
	\begin{itemize}
		\item bináris összeadás
		\item logikai mûveletek (Boole-algebra): AND, OR, XOR, NOT
		\item bitenkénti léptetés jobbra-balra (osztás, illetve szorzás 2-vel) Miért?
		\item komplemens képzés
		\item állapotjelzõk (flag regiszter bitjei) beállítása, az utasítás eredménye nulla, pozitív, negatív, elõfordult-e túlcsordulás, hiba, stb.
	\end{itemize}
	
\end{frame}

\begin{frame}{Vezérlõ egység (CU-Conrol unit)}
	Funkciói:
	
	\begin{itemize}
		\item kiolvassa a memóriából a szükséges adatokat, utasításokat
		\item értelmezi és végrehajtja az utasításokat az ALU és a mikroprogram-tár segítségével
		\item vezérli a belsõ busz adatforgalmát
		\item összehangolja a CPU többi egységének mûködését
	\end{itemize}
	
\end{frame}

\begin{frame}{Mikroprogram-tár}
	
	\begin{itemize}
		\item a processzor végre tud hajtani bizonyos egyszerûbb utasításokat hardveresen
		\item ezekbõl az elemi utasításokból a processzor gyártója elkészít (programoz) összetettebb utasításokat
		\item ezeket a programokat nevezzük mikroprogramnak
		\item ezeket tartalmazza a mikroprogram-tár
		\item a CISC processzorokban több utasítás van megvalósítva hardveresen, kevesebb a mikroprogram
		\item a RISC processzorokban kevesebb a hardveresen megvalósított utasítás, több a mikroprogram
	\end{itemize}
	
\end{frame}

\begin{frame}{A mikroprocesszor mûködése}
	
	\begin{itemize}
		\item kiolvassa a memóriából a számítógépet vezérlõ program utasításait
		\item dekódolja (értelmezi) az utasításokat
		\item vezérli és idõzíti a mûveletek elvégzéséhez szükséges adatforgalmat és a perifériák tevékenységét
		\item beolvassa a memóriából az utasítás végrehajtásához szükséges adatokat
		\item a beolvasott adatokon sorban elvégzi a szükséges mûveleteket: ezek elsõsorban logikai mûveletek lehetnek, de erre visszavezethetõk az egyéb, pl. aritmetikai mûveletek is
		\item az utasítás eredményét visszaírja a memóriába
	\end{itemize}
	
\end{frame}

\begin{frame}{Órajel és gépi ciklus}
	
	\begin{itemize}
		\item a számítógép alkatrészeinek összehangolt mûködését az órajelgenerátor biztosítja
		\item minden mûveletet ez az órajel ütemez
		\item minden gépi utasítás a  gépi ciklus egészszámú többszöröse alatt megy végbe
	\end{itemize}
	
\end{frame}

\begin{frame}{Tárolók, memória}
	
	\begin{itemize}
		\item az adatokat és az utasításokat a számítógép a memóriában tárolja
		\item a memória egysége a szó, a byte (bájt) egészszámú többszöröse
		\item minél nagyobb a szó mérete, annál nagyobb adatokkal képes a processzor egy lépésben mûveleteket végezni
		\item minden rekesznek egyedi címe, sorszáma van (fizikai cím)
		\item ennek alapján bármelyik rekesz közvetlenül elérhetõ
		\item a tárkapacitást a rendelkezésre álló rekeszek (bájtok) számával mérjük
	\end{itemize}
	
	
	\begin{table}[htbp]
		\centering
			\begin{tabular}{|ccc|}
			\hline
			1 KByte & = & 1024 bájt \\
			1 MByte & = & 1024 KByte \\
			1 GByte & = & 1024 MByte \\
			1 TByte & = & 1024 GByte \\
			\hline	
			\end{tabular}
	\end{table}
	
\end{frame}

\begin{frame}[allowframebreaks]{Tárolók osztályozása}
	
	\begin{enumerate}
		\item Az adatok elérése szerint:
		
		\begin{itemize}
			\item soros, pl. mágnesszalag
			\item közvetlen, direkt, pl. RAM (Random Access Memory)
			\item asszociatív, tartalom szerint
		\end{itemize}

		\item Az adatok módosíthatósága szerint:
		
		\begin{itemize}
			\item csak olvasható, pl. ROM (Read Only Memory)
			\item írható/olvasható, pl. RAM
			\item újraprogramozható, pl. EPROM (Erasable Programmable Read Only Memory)
		\end{itemize}
		
		\item Mûködési elv szerint:
		
		\begin{itemize}
			\item mágneses
			\item kondenzátoros
			\item optikai
		\end{itemize}
		
		\item Az adatok tárolási módja szerint:
		
		\begin{itemize}
			\item dinamikus: az áramellátás megszûnésekor törlõdik, pl. memória modulok
			\item statikus: áramellátás nélkül is megõrzi tartalmát, pl. merevlemez
		\end{itemize}
		
		\item Funkció szerint:
		
		\begin{itemize}
			\item operatív tár: gyors, de drága
			\item háttértár: olcsó, de lassú
		\end{itemize}
		
	\end{enumerate}
	
\end{frame}

\begin{frame}{Különbözõ bus (sín) rendszerek}
	
	\begin{itemize}
		\item ISA (Industry Standard Architecture)
		\begin{itemize}
			\item elõször 8, majd 16 bit
			\item 8 MHz, 6 MB/s
		\end{itemize}

		\item MCA (Micro Channel)
		\begin{itemize}
			\item 32 bit
			\item 10 MHz
		\end{itemize}

		\item EISA (Extended ISA)

\begin{frame}{Adatátviteli megoldások}
	
	\begin{description}
		\item[Programozott adatátvitel] a perifériával történõ kommunikáció a mikroprocesszor feladata (közben nem csinálhat mást)
		\item[Megszakításos adatátvitel] a mikroprocesszor közli a feladatot a perifériával, folytatja saját munkáját, a periféria megszakítással jelentkezik be ismét, ha elkészült a feladatával, vagy hiba történt
		\item[Közvetlen memória hozzáférés] ha sem az adatátvitel forrása, sem a célja nem a processzor, akkor a kommunikáció a DMA (Direct Memory Access) egység segítségével is lebonyolítható; a processzor csak definiálja a DMA feladatát, adatot nem küld és nem fogad
	\end{description}
	
\end{frame}
