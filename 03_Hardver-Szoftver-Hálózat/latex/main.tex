\subsection{Operációs rendszer}
\label{sec:OperációsRendszer}

\begin{frame}{Operációs rendszer és rendszerközeli szoftverek}
	
	\begin{itemize}
		\item \textbf{operációs rendszer}nek (rendszerszoftvernek) nevezzük azoknak a gyári programoknak az összességét, mely a számítógépes eszközök és a hálózat mûködését, az erõforrások elosztását és igénybevételét vezérlik (pl: DOS, UNIX, WINDOWS..., NOVELL NETWARE, Mac OS, stb.)
		\item \textbf{rendszerközeli szoftver}nek nevezzük azoknak a gyári programoknak összességét, melyek általános és gyakran szükséges szolgáltatásokkal az operációs rendszerhez tartozó programokat kiegészítik (pl. Total Commander)
		\item nincs éles határ a két kategória között
		\item Miért nevezzük operációs rendszernek?
	\end{itemize}
	
\end{frame}

\begin{frame}{Operációs rendszerek}


%Legelterjedtebb típusai:
\begin{itemize}
	\item \textbf{MS-DOS} (Disk Operating System, 1981-): egyfeladatos, egyfelhasználós (single user, single tasking), assembly-ben írták, parancsnyelv bázisú, interpreteres (a kapott utasítást azonnal végrehajtja, ha nem hibás a parancs)
	\item \textbf{Windows} (1985-): eredetileg egyfelhasználós, többfeladatos (multitasking), újabb változatai (XP-tõl kezdve) már többfelhasználósak (multiuser) assembly-ben és C-ben írták, grafikus bázisú (ikonok, menü, egér, \dots)
	\item \textbf{UNIX} (1973-), \textbf{Linux} (1991-): többfelhasználós, többfeladatos, C-ben írták, nyílt (nyilvános) forráskódú, parancsnyelv bázisú; a PC-ktõl a nagyszámítógépekig használt operációs rendszer.
\end{itemize}

\end{frame}

\begin{frame}{Az operációs rendszer környezete}
	
	\begin{itemize}
		\item Három csatlakozási felületet kell megvalósítani (OS $=$ Operating Sytem):
	
			\begin{enumerate}
				\item OS -- Felhasználók (kezelõk)
				\item OS -- Alkalmazások (programok)
				\item OS -- Hardver
			\end{enumerate}
	
		\item Fõ feladata virtuális gép(ek) megvalósítása és mûködtetése
		\item Mit nevezünk virtuális gépnek?

	\end{itemize}
	
\end{frame}

\begin{frame}{OS -- Felhasználók -- Felhasználói felület}
	
	\begin{itemize}
		\item Ember -- gép kapcsolat
		
		\begin{itemize}
			\item tájékoztatni kell a felhasználót a gép mûködésérõl
			\item lehetõvé kell tenni a beavatkozást
		\end{itemize}
		
		\item Kezelni kell az ehhez szükséges eszközöket
		
		\begin{itemize}
			\item képernyõ, billentyûzet, egér, hangeszközök, egyéb be-kimeneti eszközök
		\end{itemize}
		
		\item A felhasználók nem egyformák:
		
		\begin{itemize}
			\item ,,egyszerû'' felhasználók
			\item alkalmazás fejlesztõk, programozók
			\item rendszermenedzserek, rendszergazdák
		\end{itemize}
		
	\end{itemize}
	
\end{frame}

\begin{frame}[allowframebreaks]{Kezelõi felület / Egyszerû felhasználók}
	
	\begin{itemize}
		\item elsõsorban programokat futtatnak, a felhasználói programokkal kerülnek kapcsolatba, az operációs rendszerrel kevésbé
		\item a legfontosabb operációs rendszer szolgáltatás számukra a fájlrendszer
		\item bizonyos megbízhatóságot, biztonságot elvárnak, cserébe hajlandók bizonyos szabályokat követni, pl. bejelentkezés, jogosultságok, stb.
		\item számukra az operációs rendszer lehetõséget ad programok és adatok tárolására, használatára
		\framebreak
		\item az egyszerû felhasználók számára a legfontosabb operációs rendszer szolgáltatások:
		
		\begin{itemize}
			\item bejelentkezés
			\item fájlok, adatok kezelése
			\item programok áttekintése
			\item programok futtatása
			\item programok párhuzamos futtatása
			\item programok leállítása (kilövése)
		\end{itemize}
		
	\end{itemize}
	
\end{frame}

\begin{frame}{Kezelõi felület / Alkalmazásfejlesztõk}
	
	\begin{itemize}
		\item természetesen a programozók egyben ,,egyszerû'' felhasználók is, de ezen felül\dots
		\item elsõsorban programokat írnak, tesztelnek, elemeznek
		\item az operációs rendszer programozói felületét ismerik, bizonyos mértékig a belsõ mûködését is
		\item többnyire speciális fejlesztõi környezeteket használnak
		\item szükségük van az operációs rendszer szolgáltatásaira ahhoz, hogy adatokat, statisztikákat kapjanak a rendszer állapotáról (memória, erõforrások kihasználtsága, stb.)
%		\framebreak
		\item az alkalmazásfejlesztõk számára a legfontosabb operációs rendszer szolgáltatások:
		
		\begin{itemize}
			\item API-k (Application Programming Interface) a \\különbözõ rendszer erõforrások programozásához
			\item programok futtatása különbözõ módokon (debug)
		\end{itemize}
		
	\end{itemize}
	
\end{frame}

\begin{frame}{Kezelõi felület / Rendszermenedzserek}
	
	\begin{itemize}
		\item természetesen a rendszermenedzserek egyben ,,egyszerû'' felhasználók is, de ezen felül\dots
		\item Feladataik:
		
		\begin{itemize}
			\item az operációs rendszer telepítése, konfigurálása
			\item Adminisztráció: felhasználók, jogosultságok, más erõforrások
			\item Hangolás: a rendszer teljesítményét befolyásoló paraméterek figyelése és beállítása
			\item Felügyelet: hibák észlelése, elhárítása, log fájlok ellenõrzése, kezelése
			\item Biztonsági mentések rendszeres készítése
		\end{itemize}
		
		\item részletesen ismerik a rendszert
		\item olyan eszközökre van szükségük, amelyekre a többi felhasználónak nincs szüksége és õk nem is \\érhetik el azokat
		
	\end{itemize}
	
\end{frame}

\begin{frame}{Különbözõ felhasználói felületek}
	
	\begin{itemize}
		\item A felhasználók különbözõ csoportokba tartozhatnak (egyszerû felhasználó, programozó, rendszergazda)
		\item Egy felhasználó több csoportba is tartozhat
		\item A felhasználók felkészültsége, tapasztalata is nagyon eltérõ lehet
		\item Ugyanaz a felhasználó is többféle felületet igényel az eltérõ feladatokhoz
		\item Akkor milyen a jó felhasználói felület?
		
		\begin{itemize}
			\item sokféle
			\item jól paraméterezhetõ
			\item karakteres-grafikus
		\end{itemize}
		
		\item Melyik operációs rendszer tudja mindezt?
		\item Biztos, hogy egyetlen operációs rendszert kell használnunk?
		\item Hogyan használhatunk több operációs rendszert egy számítógépen?
		
	\end{itemize}
	
\end{frame}

\begin{frame}{Virtualizáció}
	
	\begin{itemize}
		\item Napjainkban az informatika talán leggyakrabban emlegetett fogalma, technológiája
		\item Nagyon sokféle dolgot értünk alatta
		\item \textit{,,A virtualizáció számítástechnikai kifejezés, számítógépi erõforrások különbözõ absztrakcióinak gyûjtõneve''}
		\item Mit lehet virtualizálni egy számítógépben?

\subsection{Hálózatok mûködése}
\label{sec:HalozatokMukodese}

\begin{frame}{Hálózati címek, IP-címek}
	
	\begin{itemize}
		\item A különbözõ hálózati rendszerek eltérõ kommunikációs és azonosítási módokat használtak
		\item Mára szinte teljesen egyeduralkodóvá vált az Internet Protocol (IP) használata
		\item Minden Internet Protocol segítségével kommunikáló eszközt egy egyedi IP-cím azonosít
		\item Kétféle IP-cím szabvány:
		
		\begin{itemize}
			\item IPv4
			
			\begin{itemize}
				\item 32 bites egész szám, 4 darab (0-255 közötti) szám pontokkal elválasztva
				\item 4 milliárd lehetséges IP-cím
			\end{itemize}
			
			\item IPv6
			
			\begin{itemize}
				\item 128 bites egész szám, 8 darab 16 bites szám kettõspontokkal elválasztva, tizenhatos számrendszerben (hexa)
				\item Hány lehetséges IP-cím van? $2^{128} = 3,2\times10^{38}$
			\end{itemize}
			
		\end{itemize}
		
	\end{itemize}
	
\end{frame}

\begin{frame}{IPv4 ,,régi'' típusú IP-címek}
	
	\begin{itemize}
		\item A 4 milliárd IP cím bõségesen elegendõnek tûnt, mára azonban elfogyóban vannak
		\item Nem adunk minden eszköznek önálló IP-címet:
	\end{itemize}
	
	
	\scriptsize{
	\begin{table}
		\centering
			\begin{tabular}{|c|c|c|c|}
			\hline
			Privát IP címtartományok & Tartomány kezdete & Tartomány vége & Címek száma \\ \hline
			24 bites tömb (/8 prefix) & 10.0.0.0 & 10.255.255.255 & 16 777 216 \\ \hline
			20 bites tömb (/12 prefix) & 172.16.0.0 & 172.31.255.255 & 1 048 576 \\ \hline
			16 bites tömb (/16 prefix) & 192.168.0.0 & 192.168.255.255 & 65 536 \\ \hline
			\end{tabular}
	\end{table}
	}

\end{frame}

\begin{frame}{IPv6 ,,új'' típusú IP-címek}
	
	\begin{itemize}
		\item 1994-ben vált szabvánnyá
		\item Gyakorlatilag korlátlan címtartomány
		\item Minden elképzelhetõ eszköz egyedi IP-címet kaphat
		\item A két rendszer párhuzamosan élhet egymás mellett
		\item Példák IPv6 címekre:
		
		\begin{itemize}
			\item fe80:0000:0000:0000:0202:b3ff:fe1e:8329
			\item fe80:0:0:0:202:b3ff:fe1e:8329, elhagyhatjuk a vezetõ nullákat
			\item fe80::202:b3ff:fe1e:8329, elhagyhatjuk az egymás melletti 0 tagokat, de csak egyszer!
			\item Most még szokatlanok ezek a számok, de meg fogunk barátkozni velük!
		\end{itemize}
		
	\end{itemize}
	
\end{frame}

\begin{frame}{Névszolgáltatás -- DNS}
	
	\begin{itemize}
		\item Az IP-címek emberi szem számára nehezen kezelhetõk
		\item Megoldás: domain nevek és DNS -- Domain Name Service
		\item Hierarchikus adatbázis
		\item Fa struktúrájú domain nevek
		\item Sok ezer szerver tárol (domain név -- IP-cím) párokat
		\item Minden szerver a maga tartományának adatait tárolja és tartja karban
		\item Ha ismerünk egy domain nevet, akkor
		
		\begin{itemize}
			\item megkérdezünk egy DNS szervert, hogy milyen IP-cím tartozik hozzá
			\item vagy tudja, és visszaadja
			\item vagy azt tudja, hogy melyik másik DNS szervertõl kell megkérdezni, és továbbítja a kérést
			\item így néhány lépés után megkapjuk a választ
			\item \textsl{a valóságban kicsit bonyolultabb a dolog, de ez az alapelv}
		\end{itemize}
		
	\end{itemize}
	
\end{frame}
