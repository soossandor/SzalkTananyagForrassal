\subsection{Logikai és aritmetikai mûveletek}
\label{sec:LogikaiMuveletek}

\begin{frame}[allowframebreaks]{Logikai mûveletek igazságtáblája}

\begin{enumerate}
	\item Negáció -- NOT (egyváltozós) ellentettjére változtatja a logikai értéket\\

\begin{table}[htbp]
	\centering
		\begin{tabular}{|c|c|}
			\hline
			$A$ & $\neg A$ \\ \hline \hline
			$0$ & $1$ \\ \hline
			$1$ & $0$ \\ \hline
		\end{tabular}
\end{table}
	
	\item És -- AND (kétváltozós) igaz, ha mindkét operandus igaz

\begin{table}[htbp]
	\centering
		\begin{tabular}{|c|c|c|}
			\hline
			$A$ & $B$ & $A \wedge B$ \\ \hline \hline
			$0$ & $0$ & $0$ \\ \hline
			$0$ & $1$ & $0$ \\ \hline
			$1$ & $0$ & $0$ \\ \hline
			$1$ & $1$ & $1$ \\ \hline
		\end{tabular}
\end{table}
	\newpage
	\item Vagy -- OR (kétváltozós) igaz, ha valamelyik operandus igaz

\begin{table}[htbp]
	\centering
		\begin{tabular}{|c|c|c|}
			\hline
			$A$ & $B$ & $A \vee B$ \\ \hline \hline
			$0$ & $0$ & $0$ \\ \hline
			$0$ & $1$ & $1$ \\ \hline
			$1$ & $0$ & $1$ \\ \hline
			$1$ & $1$ & $1$ \\ \hline
		\end{tabular}
\end{table}
\newpage

	\item Kizáró vagy -- XOR (kétváltozós) igaz, ha pontosan az egyik operandus igaz

\begin{table}[htbp]
	\centering
		\begin{tabular}{|c|c|c|}
			\hline
			$A$ & $B$ & $A\;xor\;B$ \\ \hline \hline
			$0$ & $0$ & $0$ \\ \hline
			$0$ & $1$ & $1$ \\ \hline
			$1$ & $0$ & $1$ \\ \hline
			$1$ & $1$ & $0$ \\ \hline
		\end{tabular}
\end{table}
	
\end{enumerate}

\end{frame}

\subsection{A rendszerszemlélet}
\label{sec:ARendszerszemlelet}

\begin{frame}{A rendszer fogalma}

\begin{itemize}
	\item Rendszernek nevezzük alkotóelemek és ezek kapcsolatainak olyan együttesét, amelyek az adott vizsgálat szempontjából összetartoznak
	\item A rendszerek kisebb alrendszerekbõl, és tovább nem bontható alapelemekbõl épülnek fel
	\item Alapelemnek nevezzük a rendszer azon alkotóelemeit, amelyeket az aktuális vizsgálatban nem bontunk tovább kisebb elemekre
	\item Példák: autó, számítógép, ember
	\item Vizsgáljuk meg ezeket a rendszereket!
	\item Rendszerszemlélet
\end{itemize}

\end{frame}

\begin{frame}{A rendszerek bonyolultsága}

\begin{itemize}
	\item Bonyolultság szempontjából két csoportba oszthatjuk a rendszereket:
	
	\begin{enumerate}
		\item egyszerû rendszer: 
		\begin{itemize}
			\item kisszámú alapelembõl felépülõ rendszer
			\item könnyen áttekinthetõ, vizsgálható
		\end{itemize}
		
		\item összetett rendszer: 
		\begin{itemize}
			\item nagyszámú alkotóelembõl és/vagy alrendszerekbõl felépülõ rendszer
			\item a rendszer vizsgálatához célszerû részrendszerekre bontani
		\end{itemize}
		
	\end{enumerate}
	
	\item A mérnöki rendszerek vizsgálatának alapvetõ módszere a részekre bontás
	\item Minden rendszer vizsgálható ezzel a módszerrel?
	
\end{itemize}

\end{frame}

\begin{frame}{A rendszerek bonyolultsága}

\begin{itemize}
	\item A rendszer nem egyenlõ a részeinek összegével!
	\item Minél összetettebb egy rendszer, annál nagyobb a különbség a kettõ között
	\item Mi a különbség?
	\item Kapcsolatok!!!
	\item Hogyan vizsgálhatók az összetett rendszerek?
	
	\begin{itemize}
		\item Modellezés
		\item Absztrakció
		\item Kiválasztjuk a rendszer azon jellemzõit, amelyek a vizsgálat szempontjából fontosak
		\item Felépítünk egy másik rendszert (modell), ami egyszerûbb a vizsgált rendszernél, de a kiválasztott jellemzõkben megegyezik azzal
		\item Ha jól választottuk ki a fontos jellemzõket, és jól építjük fel a modellt, akkor annak vizsgálatával fontos információkat szerezhetünk a vizsgált rendszerrõl
	\end{itemize}
	
\end{itemize}

\end{frame}

\section{A számítógép mûködése}
\label{sec:ASzamitogepMukodese}

\begin{frame}{A számítógép, mint rendszer}

\begin{itemize}
	\item A számítógép alkotóelemeit két nagy csoportba soroljuk:
	\item Hardver:
	
	\begin{itemize}
		\item hardware: kemény áru
		\item a számítógépet alkotó kézzel fogható alkatrészek összefoglaló neve
		\item elektronikus áramkörök, mechanikus eszközök, kábelek, perifériák, stb.
		\item önmagában mûködésképtelen
	\end{itemize}
	
	\item Szoftver:
	
	\begin{itemize}
		\item software: lágy áru
		\item a számítógépen futó algoritmusok, programok és adatok összessége
		\item a szoftver mûködteti a számítógép hardver eszközeit
	\end{itemize}
	
	\item Ebben a tárgyban általában a szoftverrel fogunk foglalkozni
\end{itemize}

\end{frame}

\begin{frame}{Analóg-digitális technika}

\begin{itemize}
	\item \textbf{Analóg}nak nevezzük azokat az eszközöket, eljárásokat, amelyek folytonos mennyiségekkel dolgoznak
	\item \textbf{Digitális}nak nevezzük azokat az eszközöket, eljárásokat, amelyek az adatokat diszkrét (nem folytonos) értékekkel, véges sok számjeggyel közelítik
	\item \textbf{Digitális ábrázolás} valamely változó értékének diszkrét ábrázolása véges sok számjeggyel
	\item A természetben az adatok általában analóg formában vannak jelen, hõmérséklet, áramerõsség, feszültség, színek (a fény hullámhossza)
	\item A számítógépben általában digitális adatokkal dolgozunk
	\item Szükség van az adatok átalakítására
\end{itemize}

\end{frame}

\begin{frame}{Adatok átalakítása}

\begin{enumerate}
	\item Analóg-digitális (A/D) átalakítás, A/D konverter:
	
	\begin{itemize}
		\item mintavételezés: a folytonos jelbõl véges sok helyen mintát veszünk
		\item kvantálás: a minta értéktartományát diszkrét intervallumokra osztjuk és minden intervallumot egy kijelölt elemével reprezentálunk
	\end{itemize}

	\item Digitális-analóg (D/A) átalakítás, D/A konverter:
	
	\begin{itemize}
		\item a digitális jelbõl analóg (folytonos) jelet állít elõ
	\end{itemize}
	
	\item Hol használunk minden nap analóg-digitális átalakítást mindkét irányban?\pause
	
	\begin{itemize}
		\item Modem, kábelmodem
		\item A telefon és a kábeltévé analóg jelet továbbít
	\end{itemize}
	
\end{enumerate}

\end{frame}

	\end{itemize}
	
	\item Milyen kódrendszereket ismerünk?
	
\end{itemize}

\begin{frame}{Elterjedt karakterkódolási rendszerek}

\begin{enumerate}
	\item ASCII: 
	
	\begin{itemize}
		\item 8 bites kódrendszer
		\item $2^8=256$ jelet tud megkülönböztetni
		\item eredetileg 7 bites volt, az elsõ 128 jel szabványos, egységes
		\item a második 128 karakterre több különbözõ kódkiosztás létezik
	\end{itemize}
	
	\item UNICODE:
	
	\begin{itemize}
		\item 16 bites kódrendszer
		\item $2^{16}=65536$ jelet tud megkülönböztetni
		\item ,,minden'' nyelv összes jele elfér benne
	\end{itemize}
	
	\item UTF8:
	
	\begin{itemize}
		\item Általános kódolási rendszer
		\item Minden karaktert képes kódolni 1-4 byte hosszan
		\item Változó hosszúságú kódokat használ
		\item A kódolás leírása: http://en.wikipedia.org/wiki/UTF-8
		\item Az UTF-8 kódok: http://www.utf8-chartable.de/
	\end{itemize}
	
\end{enumerate}

\end{frame}

\begin{frame}{Karakterosztályok}

\begin{itemize}
	\item alfabetikus: az angol ábécé kis és nagybetûi (a, b, c, \dots, A, B, C, \dots)
	\item numerikus: számjegyek: (0, 1, 2, \dots, 9)
	\item alfanumerikus: alfabetikus vagy numerikus, idõnként beleértünk néhány írásjelet is, pl. \_
	\item egyéb jelek: írásjelek, nemzeti karakterek, stb.
\end{itemize}

\end{frame}

\begin{frame}{Átalakítás bináris, oktális és hexadecimális számrendszer között}

\begin{itemize}
	\item korábban láttuk, hogy a számítógép számára a kettes számrendszer a legalkalmasabb
	\item a bináris számok leírása nagyon hosszú, ezért körülményesen használhatók
	\item az átalakítás tízes számrendszerrõl elég körülményes
	\item ezen a problémán segít a 8-as (oktális) és a 16-os (hexadecimális) számrendszer használata
	\item Miért?
	
	\begin{itemize}
		\item vegyük észre, hogy $8_{(10)} = 1000_{(2)}$ és $16_{(10)} = 10000_{(2)}$, azaz a 8 és a 16 ,,kerek'' számok kettes számrendszerben, ahogyan a tízes számrendszerben a száz, az ezer, vagy a millió
		\item Hogyan írnánk át egy számot tízes számrendszerbõl ezresbe?
	\end{itemize}
		
\end{itemize}

\end{frame}

\begin{frame}{Ezres tagolás}

\begin{itemize}
	\item tízes számrendszerben így szoktuk leírni a nagy számokat: $$123\ 456\ 789$$
	\item ha a háromjegyû számcsoportokat egy-egy számjegynek tekintjük, akkor ezzel átírtuk a számot ezres számrendszerbe: $$ [123]\ [456]\ [789]_{(1000)} $$

\end{itemize}

\end{frame}

	\item Zónázott BCD kód esetén minden számjegyet külön bájtban tárolunk, ez tovább csökkenti a kód tömörségét, de egyszerûbbé teszi az aritmetikát
\end{itemize}

\begin{frame}{Negatív számok kezelése}

\begin{itemize}
	\item az eddig tárgyalt módszerekkel tetszõleges pozitív egész számokat tudunk kezelni
	\item a használt bitek száma meghatározza az ábrázolható számok méretét, de bármekkora számokat megvalósíthatunk
	\item Mit tegyünk a negatív számokkal?
	
	\begin{enumerate}
		\item minden számhoz külön tároljuk az elõjelét
		
		\begin{itemize}
			\item minden alkalommal, amikor használni akarjuk a számot, meg kell vizsgálni az elõjelet, és attól függõen kell használni az értéket
		\end{itemize}
		
		\item építsük be az elõjelet a számok ábrázolásába, lehetõleg úgy, hogy jól mûködjön az aritmetika
		\begin{itemize}
			\item a számok ábrázolásához felhasználható bitek közül egyet lefoglalunk (általában a legmagasabb helyiértékût), és ezen tároljuk az elõjelet, 0: pozitív, 1: negatív
			\item az ábrázolható számok abszolútértékét ezzel megfeleztük
			\item pl. $0, \dots, +255$ helyett $-128, \dots, +127$
		\end{itemize}
		
	\end{enumerate}
	
\end{itemize}

\end{frame}

\begin{frame}{Komplemensképzés}

\begin{itemize}
	\item ezzel a módszerrel úgy ábrázoljuk a negatív számokat, hogy azokkal ugyanúgy végezhessünk mûveleteket, mint a pozitív számokkal
	\item \emph{komplemens}: az a szám, amelyik az ábrázolható legnagyobb számnál eggyel nagyobbra egészíti ki az adott számot
	\item jele: $\overline{x}$
	
	\begin{itemize}
		\item ha tízes számrendszerben háromjegyû számokkal dolgozunk
		\item a legnagyobb ábrázolható szám: $999$
		\item egy $a$ szám komplemense: $\overline{a}=1000-a$
		\item pl. $\overline{196}=1000-196=804$
	\end{itemize}
	
\end{itemize}

\end{frame}

\begin{frame}{Kettes komplemens}

\begin{itemize}
	\item Kettes számrendszerben a következõképpen képezhetjük a kettes komplemenst:
	
	\begin{enumerate}
		\item a legkisebb helyiértéktõl indulva, végigmegyünk a biteken
		\item amíg 0 biteket találunk, azokat változatlanul leírjuk
		\item amikor 1-est találunk, azt még szintén leírjuk, de a további biteket invertáljuk
	\end{enumerate}
	
	\item \emph{Egyes komplemens}: a bináris szám bitenkénti negáltja, azaz minden bitjét az ellenkezõjére állítjuk
	\item ha az egyes komplemenshez egyet hozzáadunk, megkapjuk a kettes komplemenst
\end{itemize}

\end{frame}

\end{itemize}
